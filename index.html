<script>

function formatNumberInput(el){

  let value = el.value.replace(/,/g,'');

  if(!isNaN(value) && value !== ''){

    el.value = parseInt(value).toLocaleString('en-US');

  }

}


function getNumber(id){

  return parseInt(document.getElementById(id).value.replace(/,/g,'')) || 0;

}


function calculate(){

  window.lastCalc = null; // เก็บผลลัพธ์ล่าสุดไว้ใช้ตอนทัก LINE

  let age = getNumber('age');

  let retireAge = getNumber('retireAge');

  let lifeExpect = getNumber('lifeExpect');

  let income = getNumber('income');

  let expense = getNumber('expense');

  let current = getNumber('current');


  let rateInput = parseFloat(document.getElementById('rate').value);

  let rate = isNaN(rateInput) ? 0 : rateInput / 100;


  let monthlyTarget = getNumber('annualInvest');


  let years = retireAge - age;

  let retirementYears = lifeExpect - retireAge;


  if(years <= 0 || retirementYears <= 0){

    alert("ข้อมูลไม่ถูกต้อง");

    return;

  }


  let totalNeeded = expense * 12 * retirementYears;


  let fvCurrent = rate === 0 ? current : current * Math.pow(1+rate,years);


  let monthlyRate = rate/12;

  let months = years*12;


  let fvMonthly = 0;

  if(rate === 0){

    fvMonthly = monthlyTarget * months;

  } else {

    fvMonthly = monthlyTarget * ((Math.pow(1+monthlyRate,months)-1)/monthlyRate);

  }


  let totalFuture = fvCurrent + fvMonthly;


  let gap = totalNeeded - totalFuture;


  // คำนวณเงินที่ “ต้องลงทุนจริงต่อเดือน” (PMT_needed)

  let pmtNeeded = 0;


  if(rate === 0){

    pmtNeeded = (totalNeeded - fvCurrent) / months;

  } else {

    pmtNeeded = (totalNeeded - fvCurrent) * monthlyRate / (Math.pow(1+monthlyRate,months)-1);

  }


  if(pmtNeeded < 0) pmtNeeded = 0;


  // แยกเป็น “ต้องเพิ่ม” และ “เกินเป้า” ต่อเดือน

  let additionalMonthly = Math.max(pmtNeeded - monthlyTarget, 0);

  let excessMonthly = Math.max(monthlyTarget - pmtNeeded, 0);


  // เงินที่ควรลงทุนต่อเดือนจริง

  let totalMonthlyRequired = pmtNeeded;


  // เก็บค่าไว้ใช้ตอนกดปุ่ม LINE

  window.lastCalc = {

    gap,

    additionalMonthly,

    excessMonthly,

    totalMonthlyRequired,

    income

  };


  document.getElementById('result').innerHTML = `

  <div class="box">ต้องมีเงิน <span class="highlight">${totalNeeded.toLocaleString()}</span></div>

  <div class="box">คุณจะมี <span class="highlight">${Math.round(totalFuture).toLocaleString()}</span> <span style="color:#94a3b8; font-size:13px;">(ผลจากเงินเก็บ/ลงทุนรายเดือนและเงินที่คุณมีอยู่แล้ว)</span></div>

  <div class="box">${gap >=0 ? `❗ ขาดอีก <span class="warning">${Math.round(gap).toLocaleString()}</span>` : `✅ เกินเป้า <span class="highlight">+${Math.round(Math.abs(gap)).toLocaleString()}</span>`}</div>


  <div class="box">

  👉 ต้องลงทุน <span class="highlight">${Math.round(totalMonthlyRequired).toLocaleString()}</span> / เดือน 

  <span style="color:#94a3b8; font-size:13px;">(${Math.round((totalMonthlyRequired/(income||1))*100)}% ของรายได้)</span>

  <hr>

  ${gap <= 0

    ? `🟢 คุณลงทุนเกินเป้ามา <span class="highlight">+${Math.round(excessMonthly).toLocaleString()}</span> / เดือน`

    : `❗ ยังขาดอีก <span class="warning">${Math.round(additionalMonthly).toLocaleString()}</span> / เดือน`

  }

</div>


  <div class="box" style="border:1px solid #1f2937;">

  ${gap <= 0

    ? `✨ ตอนนี้คุณมาถูกทางแล้ว แต่ยังมีวิธีที่ทำให้คุณใช้เงินน้อยลง หรือเกษียณได้เร็วขึ้นกว่านี้อีก ให้ผมช่วยวางแผนให้เหมาะกับคุณแบบเฉพาะตัว เพื่อพาคุณไปให้ถึงเป้าหมายที่ใจคุณต้องการ`

    : `⚠️ หากคุณยังไม่เริ่มวางแผนตั้งแต่วันนี้ ตัวเลขนี้มีโอกาสเพิ่มขึ้นเรื่อย ๆ ให้ผมช่วยคุณตามหาแผนเกษียณที่เหมาะกับคุณ และวางแผนให้คุณไปถึงเป้าหมายได้ง่ายขึ้น`

  }

</div>


<button onclick="contact()">ออกแบบแผนเกษียณในแบบของคุณ (ปรึกษาฟรี 15 นาที)</button>

  `;

}


function contact(){

  let base = "ผมลองคำนวณแผนเกษียณแล้ว ";

  let msg = "";


  if(window.lastCalc){

    let {gap, additionalMonthly, excessMonthly, totalMonthlyRequired, income} = window.lastCalc;


    if(gap > 0){

      msg = `ยังขาดอีก ${Math.round(additionalMonthly).toLocaleString()} บาท/เดือน และควรลงทุนรวม ${Math.round(totalMonthlyRequired).toLocaleString()} บาท/เดือน`;

    } else {

      msg = `ตอนนี้ผมลงทุนเกินเป้ามา ${Math.round(excessMonthly).toLocaleString()} บาท/เดือน (ลงทุนรวม ${Math.round(totalMonthlyRequired).toLocaleString()} บาท/เดือน)`;

    }

  } else {

    msg = "อยากขอคำแนะนำเพิ่มเติมครับ";

  }


  let finalText = encodeURIComponent(base + msg);

  window.location.href = `https://line.me/R/ti/p/@icetw?text=${finalText}`;

}
